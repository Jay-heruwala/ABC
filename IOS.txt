// Reduce

var reduceCollection1 = [10,20,30,40,50,60,70]

var sumOfReduceCollection1 = reduceCollection1.reduce(0) { $0 + $1}
print(sumOfReduceCollection1)

var multiOfReduceCollection1 = reduceCollection1.reduce(1) { $0 * $1}
print(multiOfReduceCollection1)

var findMaxNumber = reduceCollection1.reduce(0) { startNumber , number in
                                                    startNumber > number ? startNumber : number
                                                }
print(findMaxNumber)

var findNegativeNo = reduceCollection1.reduce(false) {  startNumber , number in
                                                        startNumber == true || number < 0 ? true : false
                                                     }
print(findNegativeNo)
